# INSTALLATION COMMAND:
#   cd ModelPipeline/serving/frontend
#   uv venv venv_frontend --python 3.11
#   .\venv_frontend\Scripts\Activate.ps1  # On Windows PowerShell
#   # venv_frontend\Scripts\activate.bat  # On Windows CMD
#   # source venv_frontend/bin/activate   # On Mac/Linux
#   uv pip install -r requirements.txt

# If no UV, with pip - INSTALLATION COMMAND:
#   cd ModelPipeline/serving/frontend
#   python -m venv venv_frontend
#   source venv_frontend/bin/activate  # On Windows: venv_frontend\Scripts\activate
#   pip install -r requirements.txt
#
# WHY SEPARATE ENVIRONMENT:
#   - Lightweight: ~50MB vs ~3GB (ML environment)
#   - Fast installs: 30 seconds vs 5+ minutes
#   - Clean separation: Frontend doesn't need torch, transformers, etc.
#   - Portable: Can deploy frontend independently (e.g., Streamlit Cloud)
#
# STREAMLIT NATIVE FEATURES (No extra packages needed):
#   - Chat interface: st.chat_message(), st.chat_input()
#   - Layout: st.columns(), st.container(), st.expander()
#   - Metrics: st.metric()
#   - Theming: .streamlit/config.toml
#
